## SQL Select

### Basic select

```typescript
const result = await db.select().from(users);
// SELECT "id", "name", "age" FROM "users";
```

### Partial select

```typescript
const result = await db.select({
  field1: users.id,
  field2: users.name,
}).from(users);

// With SQL expressions
const result = await db.select({
  id: users.id,
  lowerName: sql<string>`lower(${users.name})`,
}).from(users);
```

### Conditional select

```typescript
async function selectUsers(withName: boolean) {
  return db.select({
    id: users.id,
    ...(withName ? { name: users.name } : {}),
  }).from(users);
}
```

### Distinct

```typescript
await db.selectDistinct().from(users).orderBy(users.id);

// PostgreSQL: DISTINCT ON
await db.selectDistinctOn([users.id]).from(users).orderBy(users.id);
```

### Filters (WHERE)

```typescript
import { eq, ne, gt, gte, lt, lte, and, or, not, isNull, isNotNull,
  inArray, notInArray, between, notBetween, like, notLike, ilike,
  exists, notExists } from "drizzle-orm";

// Simple
db.select().from(users).where(eq(users.id, 42));
db.select().from(users).where(ne(users.name, "Dan"));
db.select().from(users).where(gt(users.age, 18));
db.select().from(users).where(gte(users.age, 18));
db.select().from(users).where(lt(users.age, 65));
db.select().from(users).where(lte(users.age, 65));

// Null checks
db.select().from(users).where(isNull(users.name));
db.select().from(users).where(isNotNull(users.name));

// Array
db.select().from(users).where(inArray(users.id, [1, 2, 3]));
db.select().from(users).where(notInArray(users.id, [1, 2, 3]));

// Range
db.select().from(users).where(between(users.age, 18, 65));
db.select().from(users).where(notBetween(users.age, 18, 65));

// Pattern matching
db.select().from(users).where(like(users.name, "%john%"));
db.select().from(users).where(notLike(users.name, "%john%"));
db.select().from(users).where(ilike(users.name, "%john%")); // case-insensitive (PG)

// Combining
db.select().from(users).where(and(eq(users.id, 42), eq(users.name, "Dan")));
db.select().from(users).where(or(eq(users.id, 42), eq(users.name, "Dan")));
db.select().from(users).where(not(eq(users.id, 42)));

// Column-to-column comparison
db.select().from(table).where(eq(table.col1, table.col2));
```

### Advanced filters (conditional)

```typescript
const searchPosts = async (term?: string) => {
  return db.select().from(posts).where(term ? ilike(posts.title, term) : undefined);
};

// Dynamic filter array
const filters: SQL[] = [];
filters.push(ilike(posts.title, "AI"));
filters.push(inArray(posts.category, ["Tech", "Art"]));
filters.push(gt(posts.views, 200));
await db.select().from(posts).where(and(...filters));
```

### Array operators (PostgreSQL)

```typescript
import { arrayContains, arrayContained, arrayOverlaps } from "drizzle-orm";

arrayContains(users.interests, ["music"]);       // ARRAY CONTAINS
arrayContained(users.interests, ["music", "art"]); // ARRAY CONTAINED BY
arrayOverlaps(users.interests, ["music"]);        // ARRAY OVERLAPS
```

### Limit & offset


### Order by

```typescript
await db.select().from(users).orderBy(users.id);
```

### Aggregations

```typescript
import { count, sum, avg, min, max } from "drizzle-orm";

await db.select({ count: count() }).from(users);
await db.select({ totalAge: sum(users.age) }).from(users);
await db.select({ avgAge: avg(users.age) }).from(users);
await db.select({ minAge: min(users.age), maxAge: max(users.age) }).from(users);

// Group by
await db.select({ city: users.city, count: count() })
  .from(users).groupBy(users.city);

// Having
await db.select({ city: users.city, count: count() })
  .from(users).groupBy(users.city).having(gt(count(), 10));
```

## SQL Insert

### Basic insert

```typescript
await db.insert(users).values({ name: "John", age: 30 });
```

### Multiple rows

```typescript
await db.insert(users).values([
  { name: "Andrew", age: 25 },
  { name: "Dan", age: 30 },
]);
```

### Returning (PostgreSQL)

```typescript
const result = await db.insert(users).values({ name: "Dan" }).returning();
const partial = await db.insert(users)
  .values({ name: "Dan" })
  .returning({ insertedId: users.id });
```

### Upserts and conflicts

```typescript
import { onConflictDoNothing, onConflictDoUpdate } from "drizzle-orm/pg-core";

// On conflict do nothing
await db.insert(users)
  .values({ id: 1, name: "Dan" })
  .onConflictDoNothing();

// On conflict do nothing with target
await db.insert(users)
  .values({ id: 1, name: "Dan" })
  .onConflictDoNothing({ target: users.id });

// On conflict do update
await db.insert(users)
  .values({ id: 1, name: "Dan" })
  .onConflictDoUpdate({
    target: users.id,
    set: { name: "Dan" },
  });
```

### Insert from select

```typescript
await db.insert(users)
  .select(db.select({ name: oldUsers.name }).from(oldUsers));
```

### WITH clause

```typescript
const cte = db.$with("cte").as(db.select().from(users));
await db.with(cte).insert(users).select(db.select().from(cte));
```

## Joins

### Join types

```typescript
// Left join
const leftResult = await db.select()
  .from(users)
  .leftJoin(pets, eq(users.id, pets.ownerId));

// Right join
const rightResult = await db.select()
  .from(users)
  .rightJoin(pets, eq(users.id, pets.ownerId));

// Inner join
const innerResult = await db.select()
  .from(users)
  .innerJoin(pets, eq(users.id, pets.ownerId));

// Full join (PostgreSQL)
const fullResult = await db.select()
  .from(users)
  .fullJoin(pets, eq(users.id, pets.ownerId));

// Cross join
const crossResult = await db.select()
  .from(users)
  .crossJoin(pets);
```

### Lateral joins (PostgreSQL)

```typescript
const subquery = db.select()
  .from(pets)
  .where(gte(users.age, 16))
  .as("userPets");

const result = await db.select()
  .from(users)
  .leftJoinLateral(subquery, sql`true`);
```

### Partial select with joins

```typescript
const result = await db.select({
  id: users.id,
  name: users.name,
  petName: pets.name,
})
  .from(users)
  .leftJoin(pets, eq(users.id, pets.ownerId));
```

### Aliases & self-joins

```typescript
import { alias } from "drizzle-orm/pg-core";

const parent = alias(user, "parent");
const result = await db.select()
  .from(user)
  .leftJoin(parent, eq(parent.id, user.parentId));
```

### Aggregating join results

```typescript
type User = typeof users.$inferSelect;
type Pet = typeof pets.$inferSelect;

const rows = await db.select({ user: users, pet: pets })
  .from(users)
  .leftJoin(pets, eq(users.id, pets.ownerId));

const result = rows.reduce<Record<number, { user: User; pets: Pet[] }>>(
  (acc, row) => {
    if (!acc[row.user.id]) acc[row.user.id] = { user: row.user, pets: [] };
    if (row.pet) acc[row.user.id]!.pets.push(row.pet);
    return acc;
  },
  {}
);
```

## Relational Queries API (RQB)

### Setup

Define relations separately from schema:

```typescript
import { defineRelations } from "drizzle-orm";
import { users, posts, comments } from "./schema";

export const relations = defineRelations({ users, posts, comments }, (r) => ({
  users: {
    posts: r.many.posts(),
  },
  posts: {
    author: r.one.users({ from: r.posts.authorId, to: r.users.id }),
    comments: r.many.comments(),
  },
  comments: {
    post: r.one.posts({ from: r.comments.postId, to: r.posts.id }),
  },
}));
```

Pass relations to drizzle:

```typescript
import { relations } from "./relations";
const db = drizzle(process.env.DATABASE_URL!, { relations });
```

```typescript
const users = await db.query.users.findMany();
```

## Drizzle Relations

### one() â€” single relation

```typescript
const relations = defineRelations({ users, posts }, (r) => ({
  posts: {
    author: r.one.users({
      from: r.posts.ownerId,    // Source column
      to: r.users.id,           // Target column
      optional: false,          // Required (non-null) in results
      alias: "custom_name",     // For disambiguating multiple relations
      where: { verified: true }, // Filter (polymorphic)
    }),
  },
}));
```

### many() â€” collection relation

```typescript
const relations = defineRelations({ users, posts }, (r) => ({
  users: {
    feed: r.many.posts({
      from: r.users.id,
      to: r.posts.ownerId,
      alias: "custom_name",
      where: { approved: true },
    }),
  },
}));
```

### One-to-one

```typescript

  profile: r.one.profiles({ from: r.users.id, to: r.profiles.userId }),
},
```

### One-to-many

```typescript
users: { posts: r.many.posts() },

```

### Many-to-many (through junction)

```typescript
export const relations = defineRelations({ users, groups, usersToGroups }, (r) => ({
  users: {
    groups: r.many.groups({
      from: r.users.id.through(r.usersToGroups.userId),
      to: r.groups.id.through(r.usersToGroups.groupId),

  },
  groups: {
    participants: r.many.users(),
  },

```

```typescript

  editor: r.one.users({ from: r.posts.editorId, to: r.users.id, alias: "editor" }),
},
```

## sql Template Tag

### Raw SQL

```typescript
import { sql } from "drizzle-orm";

const id = 69;
await db.execute(sql`select * from ${usersTable} where ${usersTable.id} = ${id}`);
// SELECT * FROM "users" WHERE "users"."id" = $1; -- [69]
```

Tables and columns are auto-escaped. Dynamic params become parameterized values (SQL injection safe).

### Type annotation

```typescript
// Without type â€” inferred as unknown
const r1 = await db.select({ lower: sql`lower(${users.name})` }).from(users);

// With type â€” inferred as string
const r2 = await db.select({ lower: sql<string>`lower(${users.name})` }).from(users);
```

### mapWith â€” runtime mapping

```typescript
// Map like a column type
sql`...`.mapWith(users.name);

// Custom mapping
sql`...`.mapWith({
  mapFromDriverValue: (value: any) => value.toUpperCase(),
});

// Map with constructor
sql`...`.mapWith(Number);
```

### as â€” alias

```typescript
sql`lower(${users.name})`.as("lower_name");
// lower("users"."name") as lower_name
```

### sql.raw, sql.empty, sql.append, sql.join

```typescript
sql.raw("NOW()");                    // Raw SQL without parameterization
sql.empty();                         // Empty SQL fragment
sql`SELECT `.append(sql`1`);         // Append fragments
sql.join([sql`a`, sql`b`], sql`, `); // Join with separator
```

## Transactions

```typescript
await db.transaction(async (tx) => {
  await tx.update(accounts)
    .set({ balance: sql`${accounts.balance} - 100.00` })
    .where(eq(users.name, "Dan"));

  await tx.update(accounts)
    .set({ balance: sql`${accounts.balance} + 100.00` })
    .where(eq(users.name, "Andrew"));
}, {
  isolationLevel: "read committed",  // "read uncommitted" | "read committed" | "repeatable read" | "serializable"
  accessMode: "read write",           // "read only" | "read write"
  deferrable: true,
});
```

## Batch API

Execute multiple SQL statements in a single round trip (Neon HTTP driver for PostgreSQL):

```typescript
const batchResponse = await db.batch([
  db.insert(usersTable).values({ id: 1, name: "John" }).returning({ id: usersTable.id }),
  db.update(usersTable).set({ name: "Dan" }).where(eq(usersTable.id, 1)),
  db.query.usersTable.findMany({}),
  db.select().from(usersTable).where(eq(usersTable.id, 1)),
]);

// batchResponse is a tuple with typed results for each query
// type: [{ id: number }[], NeonHttpQueryResult, {...}[], {...}[], {...}[]]
```

Supported builders inside `db.batch()`:
- `db.query.<table>.findMany()`
- `db.query.<table>.findFirst()`
- `db.select()...`
- `db.update()...`
- `db.delete()...`
- `db.insert()...`

## Cache

Drizzle provides a built-in query cache with Upstash Redis integration.

### Quickstart (Upstash)

```typescript
import { upstashCache } from "drizzle-orm/cache/upstash";
import { drizzle } from "drizzle-orm/...";

const db = drizzle(process.env.DB_URL!, {
  cache: upstashCache({
    url: process.env.UPSTASH_URL,
    token: process.env.UPSTASH_TOKEN,
    global: true,        // Enable caching for ALL queries by default
    config: { ex: 60 },  // Default TTL: 60 seconds
  }),
});
```

### Opt-in caching (global: false, default)

```typescript
// Queries won't read from cache unless explicitly opted in
const res = await db.select().from(users);

// Opt in with .$withCache()
const cached = await db.select().from(users).$withCache();

// Custom options
const cached = await db.select().from(users).$withCache({
  config: {},                // Override cache config for this query
  tag: "custom_key",         // Custom cache key
  autoInvalidate: false,     // Disable auto-invalidation (eventual consistency)
});
```

### Global caching (global: true)

```typescript
// All queries read from cache automatically
const res = await db.select().from(users);

// Disable cache for a specific query
const fresh = await db.select().from(users).$withCache(false);
```

### Cache invalidation

Mutation operations (insert/update/delete) automatically trigger `onMutate` to invalidate cached queries involving affected tables. Use `autoInvalidate: false` for manual control (e.g., data that doesn't change often).

## Dynamic Query Building

By default, Drizzle query builders only allow calling methods like `.where()` once. For dynamic query construction, use `.$dynamic()`:

```typescript
function withPagination<T extends PgSelect>(qb: T, page: number = 1, pageSize: number = 10) {
  return qb.limit(pageSize).offset((page - 1) * pageSize);
}

// âťŚ Type error â€” not in dynamic mode
const query = db.select().from(users).where(eq(users.id, 1));
withPagination(query, 1);

// âś… OK â€” dynamic mode
const dynamicQuery = query.$dynamic();
withPagination(dynamicQuery, 1);
```

Dynamic mode allows calling methods multiple times and modifying the query builder generically:

```typescript
function withFriends<T extends PgSelect>(qb: T) {
  return qb.leftJoin(friends, eq(friends.userId, users.id));
}

let query = db.select().from(users).where(eq(users.id, 1)).$dynamic();
query = withFriends(query);
```

Dynamic query builder types: `PgSelect`, `PgSelectQueryBuilder`, `PgInsert`, `PgUpdate`, `PgDelete` (and equivalents for MySQL/SQLite).

## Read Replicas

Distribute SELECT queries across read replicas, route writes to primary:

```typescript
import { withReplicas } from "drizzle-orm/pg-core";

const primaryDb = drizzle("postgres://user:password@host:port/primary_db");
const read1 = drizzle("postgres://user:password@host:port/read_replica_1");
const read2 = drizzle("postgres://user:password@host:port/read_replica_2");

const db = withReplicas(primaryDb, [read1, read2]);

// SELECT â†’ routed to a read replica
await db.select().from(usersTable);

// DELETE/INSERT/UPDATE â†’ routed to primary
await db.delete(usersTable).where(eq(usersTable.id, 1));

// Force read from primary
await db.$primary.select().from(usersTable);
```

### Custom replica selection

```typescript
const db = withReplicas(primaryDb, [read1, read2], (replicas) => {
  const weight = [0.7, 0.3]; // 70% to read1, 30% to read2
  let cumulativeProbability = 0;
  const rand = Math.random();
  for (const [i, replica] of replicas.entries()) {
    cumulativeProbability += weight[i]!;
    if (rand < cumulativeProbability) return replica;
  }
  return replicas[0]!;
});
```

## Custom Types

Define your own column types using `customType()`:

```typescript
import { customType, pgTable } from "drizzle-orm/pg-core";

const customInteger = customType<{ data: number }>({
  dataType() { return "integer"; },
});

const customText = customType<{ data: string }>({
  dataType() { return "text"; },
});

// With config, fromDriver, and codec
const customTimestamp = customType<{
  data: Date;
  driverData: string;
  config: { withTimezone: boolean; precision?: number };
  configRequired: false;
}>({
  dataType(config) {
    const precision = typeof config?.precision !== "undefined" ? ` (${config.precision})` : "";
    return `timestamp${precision}${config?.withTimezone ? " with time zone" : ""}`;
  },
  fromDriver(value: string): Date {
    return new Date(value);
  },
});

export const users = pgTable("users", {
  id: customInteger().primaryKey(),
  name: customText().notNull(),
  createdAt: customTimestamp("created_at", { withTimezone: true }).notNull().default(sql`now()`),
});
```

### Transform layers and execution order

- **Writes (Insert/Update):** `toDriver` first â†’ codec `normalizeParam` second
- **Reads (Select):** codec `normalize` first â†’ `fromDriver` second
- If no codec is set, only `toDriver`/`fromDriver` run
- If no `toDriver`/`fromDriver` is set, only codecs run (if defined)
- If neither is set, values pass through untouched

## Codecs

Codecs are driver-level transform layers with two sub-layers: **cast** (SQL level) and **normalize** (JS level).

### How codecs work

```
Reads:  SQL cast â†’ DB returns text â†’ JS normalize â†’ your code
Writes: JS value â†’ JS normalize â†’ SQL cast â†’ DB receives typed value
```

Each layer operates across 3 contexts (regular, inside JSON, params), each with scalar and array variants.

### Overriding codecs for built-in types

```typescript
const db = drizzle(client, {
  codecs: {
    "bigint:number": {
      cast: (name) => sql`${name}::text`,
      normalize: BigInt,
    },
  },
});
```

### Codecs in customType()

```typescript
// Use existing codec by name
const customBigint = customType<{ data: bigint; driverData: string }>({
  dataType() { return "bigint"; },
  codec: "bigint",
});

// Dynamic codec based on config
const customDataType = customType<{
  data: number;
  driverData: bigint | number;
  config?: { mode: "bigint" | "number" };
  configRequired: true;
}>({
  dataType() { return "custom_type"; },
  codec: (config) => config?.mode === "number" ? "bigint:number" : "bigint",
});

// No codec â€” values pass through as-is
const rawCustom = customType<{ data: string; driverData: string }>({
  dataType() { return "my_type"; },
});
```

Codec field accepts: a string (codec name), a function `(config) => string | undefined`, or `undefined` (default, no codec).

## JIT Mappers

JIT (Just-In-Time) mappers are dynamically generated JavaScript functions that transform database rows into JS objects. Instead of interpreting column metadata at runtime for every row, JIT mappers compile a specialized function once and reuse it for all rows in the result set.

**Without JIT mappers:** For each row, for each column â†’ look up decoder, check codec, check null, apply codec, apply decoder, navigate path, assign to object.

**With JIT mappers:** All metadata lookups, null checks, and nested object construction are hardcoded into a generated function at compile time, eliminating interpretive overhead.

## Views

### Declaring views (PostgreSQL)

```typescript
import { pgView, pgTable, serial, text, timestamp } from "drizzle-orm/pg-core";
import { eq } from "drizzle-orm";

export const user = pgTable("user", {
  id: serial().primaryKey(),
  name: text(),
  role: text().$type<"admin" | "customer">(),
});

// Query builder syntax
export const userView = pgView("user_view").as((qb) => qb.select().from(user));
export const customersView = pgView("customers_view").as((qb) =>
  qb.select().from(user).where(eq(user.role, "customer"))
);

// Raw SQL syntax
export const newYorkers = pgView("new_yorkers", {
  id: serial("id").primaryKey(),
  name: text("name").notNull(),
}).as(sql`select * from ${users} where ${eq(users.cityId, 1)}`);
```

### Declaring existing views

For read-only access to existing database views (Drizzle Kit won't generate CREATE VIEW):

```typescript
export const trimmedUser = pgView("trimmed_user", {
  id: serial("id"),
  name: text("name"),
  email: text("email"),
}).existing();
```

### Materialized views (PostgreSQL)

```typescript
import { pgMaterializedView } from "drizzle-orm/pg-core";

export const newYorkers = pgMaterializedView("new_yorkers").as((qb) =>
  qb.select().from(users).where(eq(users.cityId, 1))
);

// Refresh at runtime
await db.refreshMaterializedView(newYorkers);
await db.refreshMaterializedView(newYorkers).concurrently();
await db.refreshMaterializedView(newYorkers).withNoData();
```

## Indexes & Constraints

### Unique constraints

```typescript
import { integer, text, unique, pgTable } from "drizzle-orm/pg-core";

// Column-level
export const user = pgTable("user", { id: integer().unique() });

// Named
export const table = pgTable("table", { id: integer().unique("custom_name") });

// Composite
export const composite = pgTable("composite_example", {
  id: integer(), name: text(),
}, (t) => [
  unique().on(t.id, t.name),
  unique("custom_name").on(t.id, t.name),
]);

// NULLS NOT DISTINCT (Postgres 15+)
export const userNulls = pgTable("user_nulls", {
  id: integer(),
}, (t) => [ unique().on(t.id).nullsNotDistinct() ]);
```

### Check constraints

```typescript
import { check, integer, pgTable, text, uuid } from "drizzle-orm/pg-core";

export const users = pgTable("users", {
  id: uuid().defaultRandom().primaryKey(),
  username: text().notNull(),
  age: integer(),
}, (table) => [
  check("age_check", sql`${table.age} > 21`),
]);
```

### Foreign keys

```typescript
// Inline
export const book = pgTable("book", {
  id: serial().primaryKey(),
  authorId: integer("author_id").references(() => user.id),
});

// Self-reference
export const user = pgTable("user", {
  id: serial().primaryKey(),
  parentId: integer("parent_id").references((): AnyPgColumn => user.id),
});

// Composite foreign key with custom name
foreignKey({
  columns: [table.userFirstName, table.userLastName],
  foreignColumns: [user.firstName, user.lastName],
  name: "custom_fk",
})
```

### Indexes

```typescript
import { index, uniqueIndex, pgTable } from "drizzle-orm/pg-core";

export const user = pgTable("user", {
  id: serial().primaryKey(),
  name: text(),
  email: text(),
}, (table) => [
  index("name_idx").on(table.name),
  uniqueIndex("email_idx").on(table.email),
]);

// Advanced index options
index("name_idx")
  .on(table.column1.asc(), table.column2.nullsFirst())
  .concurrently()
  .where(sql`...`)
  .with({ fillfactor: "70" });

// Using specific index type
index("name_idx")
  .using("btree", table.column1.asc(), sql`lower(${table.column2})`)
  .concurrently()
  .where(sql`...`);
```

## Generated Columns

Stored generated columns are computed on insert/update and persisted in the database:

```typescript
// SQL expression
export const test = pgTable("test", {
  generatedName: text("gen_name").generatedAlwaysAs(sql`'hello "world"!'`),
});

// Callback referencing other columns
export const test = pgTable("test", {
  name: text("first_name"),
  generatedName: text("gen_name").generatedAlwaysAs(
    (): SQL => sql`'hi, ' || ${test.name} || '!'`
  ),
});

// Full-text search example with custom tsvector type
import { customType, index, integer, pgTable, text } from "drizzle-orm/pg-core";

const tsVector = customType<{ data: string }>({
  dataType() { return "tsvector"; },
});

export const test = pgTable("test", {
  id: integer("id").primaryKey().generatedAlwaysAsIdentity(),
  content: text("content"),
  contentSearch: tsVector("content_search").generatedAlwaysAs(
    (): SQL => sql`to_tsvector('english', ${test.content})`
  ),
}, (t) => [
  index("idx_content_search").using("gin", t.contentSearch),
]);
```

**Limitations:**
- STORED only (no VIRTUAL)
- Cannot specify default values
- Expressions cannot reference other generated columns or include subqueries
- Cannot use in primary keys, foreign keys, or unique constraints

## Set Operations

Drizzle supports SQL set operations: `union`, `unionAll`, `intersect`, `intersectAll`, `except`, `exceptAll`.

### Union (remove duplicates)

```typescript
import { union } from "drizzle-orm/pg-core";

// Function syntax
const result = await union(
  db.select({ name: users.name }).from(users),
  db.select({ name: customers.name }).from(customers)
).limit(10);

// Method syntax
const result = await db.select({ name: users.name }).from(users)
  .union(db.select({ name: customers.name }).from(customers))
  .limit(10);
```

### Union All (keep duplicates)

```typescript
import { unionAll } from "drizzle-orm/pg-core";

const result = await unionAll(
  db.select({ transaction: onlineSales.transactionId }).from(onlineSales),
  db.select({ transaction: inStoreSales.transactionId }).from(inStoreSales)
);
```

### Intersect (common rows, no duplicates)

```typescript
import { intersect } from "drizzle-orm/pg-core";

const result = await intersect(
  db.select({ courseName: depA.courseName }).from(depA),
  db.select({ courseName: depB.courseName }).from(depB)
);
```

### Intersect All (common rows, with duplicates)

```typescript
import { intersectAll } from "drizzle-orm/pg-core";

const result = await intersectAll(
  db.select({ productId: regularCustomerOrders.productId }).from(regularCustomerOrders),
  db.select({ productId: vipCustomerOrders.productId }).from(vipCustomerOrders)
);
```

### Except (rows in A not in B, no duplicates)

```typescript
import { except } from "drizzle-orm/pg-core";

const result = await except(
  db.select({ projectsName: depA.projectsName }).from(depA),
  db.select({ projectsName: depB.projectsName }).from(depB)
);
```

### Except All (rows in A not in B, with duplicates)

```typescript
import { exceptAll } from "drizzle-orm/pg-core";

const result = await exceptAll(
  db.select({ productId: regularCustomerOrders.productId }).from(regularCustomerOrders),
  db.select({ productId: vipCustomerOrders.productId }).from(vipCustomerOrders)
);
```

All set operations also support method chaining syntax: `.union()`, `.unionAll()`, `.intersect()`, `.intersectAll()`, `.except()`, `.exceptAll()`.

## WITH Clause (CTEs)

Common Table Expressions (CTEs) using `db.$with()` and `db.with()`:

```typescript
// Basic CTE
const sq = db.$with("sq").as(db.select().from(users).where(eq(users.id, 42)));
const result = await db.with(sq).select().from(sq);

// CTE with INSERT (returning)
const sq = db.$with("sq").as(
  db.insert(users).values({ name: "John" }).returning()
);
const result = await db.with(sq).select().from(sq);

// CTE with UPDATE (returning)
const sq = db.$with("sq").as(
  db.update(users).set({ age: 25 }).where(eq(users.name, "John")).returning()
);
const result = await db.with(sq).select().from(sq);

// CTE with DELETE (returning)
const sq = db.$with("sq").as(
  db.delete(users).where(eq(users.name, "John")).returning()
);
const result = await db.with(sq).select().from(sq);
```

**Aliasing SQL expressions in CTEs** â€” must use `.as()` to reference fields later:

```typescript
const sq = db.$with("sq").as(
  db.select({ name: sql<string>`upper(${users.name})`.as("name") }).from(users)
);
const result = await db.with(sq).select({ name: sq.name }).from(sq);
```

Without an alias, the field type becomes `DrizzleTypeError` and cannot be referenced.
