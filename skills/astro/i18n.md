# Internationalization (i18n)

Astro provides built-in i18n routing and configuration for multi-language sites.

---

## Configuration

Configure i18n in `astro.config.mjs`:

```ts
import { defineConfig } from 'astro/config';

export default defineConfig({
  i18n: {
    defaultLocale: 'en',
    locales: ['en', 'fr', 'es', 'de'],
    routing: {
      prefixDefaultLocale: true, // /en/, /fr/, /es/
      // prefixDefaultLocale: false, // /, /fr/, /es/ (no prefix for default)
    },
    fallback: {
      es: 'en', // fallback to English for missing Spanish pages
    },
  },
});
```

### i18n Config Options

| Option | Type | Description |
|--------|------|-------------|
| `defaultLocale` | `string` | Default language code |
| `locales` | `string[]` | Supported locale codes |
| `routing.prefixDefaultLocale` | `boolean` | Prefix default locale in URL |
| `fallback` | `Record<string, string>` | Fallback locale mapping |
| `routing.redirectToDefaultLocale` | `boolean` | Redirect `/` to default locale |
| `domains` | `Record<string, string>` | Custom domains per locale |

## Folder Structure

### With `prefixDefaultLocale: true`

```
src/pages/
  en/
    index.astro        → /en/
    about.astro        → /en/about
  fr/
    index.astro        → /fr/
    about.astro        → /fr/about
  es/
    index.astro        → /es/
    about.astro        → /es/about
```

### With `prefixDefaultLocale: false`

```
src/pages/
  index.astro          → /         (default locale, no prefix)
  about.astro          → /about
  fr/
    index.astro        → /fr/
    about.astro        → /fr/about
  es/
    index.astro        → /es/
    about.astro        → /es/about
```

## Translating Routes

Different URL slugs per locale:

```
src/pages/
  en/about.astro       → /en/about
  fr/a-propos.astro    → /fr/a-propos
  es/acerca.astro      → /es/acerca
```

## Content Collections for Translated Content

```ts
// src/content.config.ts
import { defineCollection, z } from 'astro:content';
import { glob } from 'astro/loaders';

const blog = defineCollection({
  loader: glob({ pattern: '**/*.md', base: './src/content/blog' }),
  schema: z.object({
    title: z.string(),
    lang: z.enum(['en', 'fr', 'es']),
    pubDate: z.coerce.date(),
  }),
});

export const collections = { blog };
```

```
src/content/blog/
  en/
    hello-world.md
  fr/
    bonjour-le-monde.md
  es/
    hola-mundo.md
```

### Querying by Language

```astro
---
import { getCollection } from 'astro:content';
const { lang } = Astro.params;
const posts = await getCollection('blog', (entry) => entry.data.lang === lang);
---
```

## Translating UI Strings

Create a translations file:

```ts
// src/i18n/ui.ts
export const languages = {
  en: 'English',
  fr: 'Français',
  es: 'Español',
};

export const defaultLang = 'en';

export const ui = {
  en: {
    'nav.home': 'Home',
    'nav.about': 'About',
    'nav.blog': 'Blog',
    'footer.copyright': '© 2024',
  },
  fr: {
    'nav.home': 'Accueil',
    'nav.about': 'À propos',
    'nav.blog': 'Blog',
    'footer.copyright': '© 2024',
  },
  es: {
    'nav.home': 'Inicio',
    'nav.about': 'Acerca de',
    'nav.blog': 'Blog',
    'footer.copyright': '© 2024',
  },
};
```

### Translation Helper

```ts
// src/i18n/utils.ts
import { ui, defaultLang } from './ui';

export function getLangFromUrl(url: URL) {
  const [, lang] = url.pathname.split('/');
  if (lang in ui) return lang as keyof typeof ui;
  return defaultLang;
}

export function useTranslations(lang: keyof typeof ui) {
  return function t(key: keyof (typeof ui)[typeof defaultLang]) {
    return ui[lang][key] || ui[defaultLang][key];
  };
}
```

### Using in Components

```astro
---
import { getLangFromUrl, useTranslations } from '../i18n/utils';
const lang = getLangFromUrl(Astro.url);
const t = useTranslations(lang);
---
<nav>
  <a href={`/${lang}/`}>{t('nav.home')}</a>
  <a href={`/${lang}/about`}>{t('nav.about')}</a>
  <a href={`/${lang}/blog`}>{t('nav.blog')}</a>
</nav>
```

## Language Switcher

```astro
---
import { languages } from '../i18n/ui';
const currentLang = getLangFromUrl(Astro.url);
const pathWithoutLang = Astro.url.pathname.replace(`/${currentLang}`, '');
---
<div class="lang-switcher">
  {Object.entries(languages).map(([code, name]) => (
    <a
      href={code === 'en' && !prefixDefaultLocale
        ? pathWithoutLang
        : `/${code}${pathWithoutLang}`
      }
      class:list={{ active: code === currentLang }}
    >
      {name}
    </a>
  ))}
</div>
```

## Hiding Default Language in URL

With `prefixDefaultLocale: false`:

```ts
export default defineConfig({
  i18n: {
    defaultLocale: 'en',
    locales: ['en', 'fr', 'es'],
    routing: {
      prefixDefaultLocale: false,
    },
  },
});
```

- `/` serves English content (no `/en/` prefix)
- `/fr/` and `/es/` are prefixed as normal

## Custom Domains per Locale

```ts
export default defineConfig({
  i18n: {
    defaultLocale: 'en',
    locales: ['en', 'fr'],
    domains: {
      en: 'https://example.com',
      fr: 'https://example.fr',
    },
  },
});
```

## Astro i18n API

### `Astro.currentLocale`

```astro
---
const locale = Astro.currentLocale; // 'en', 'fr', etc.
---
```

### `Astro.preferredLocale`

```astro
---
const preferred = Astro.preferredLocale; // from Accept-Language header
---
```

### `Astro.preferredLocaleList`

```astro
---
const locales = Astro.preferredLocaleList; // array of preferred locales
---
```
